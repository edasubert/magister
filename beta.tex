\section{12-fold rotational symmetry}
\subsection{Definition}
\begin{align*}
x^2 &=4x-1 &
\beta_{12} &= 2 + \sqrt{3} \doteq 3.732 &
\beta_{12}' &= 2 - \sqrt{3} \doteq 0.268 \\
\end{align*}

\subsection{Formulas}
\begin{align*}
\beta_{12}\beta_{12}' &= 1 &
\beta_{12} + \beta_{12}' &= 4 &
\beta_{12}^{k+2} &= 4\beta_{12}^{k+1} - \beta_{12}^k \\
\end{align*}

\subsection{Distances}
\begin{remark}
$\frac{1}{\beta_{12}} = 4-\beta_{12}$, $\frac{\beta_{12} - 2}{\beta_{12}} = 2\beta_{12}-7$, $\frac{\beta_{12} - 1}{\beta_{12}} = \beta_{12}-3$
\end{remark}
