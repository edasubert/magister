\section{8-fold rotational symmetry}
\subsection{Definition}
\begin{align*}
x^2 &=2x+1 &
\beta_8 &= 1 + \sqrt{2} \doteq 2.414 &
\beta_8' &= 1 - \sqrt{2} \doteq -0.414 \\
\end{align*}

\subsection{Formulas}
\begin{align*}
\beta_8\beta_8' &= -1 &
\beta_8 + \beta_8' &= 2 &
\beta_8^{k+2} &= 2\beta_8^{k+1} + \beta_8^k \\
\end{align*}
